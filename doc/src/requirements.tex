\section{Requirements}

\subsection{Functional requirements}
\begin{table}[h!]
    \begin{ReqTable}
    \ReqHeader
    \ReqItem{R:Battery}{The source is internally powered. The internal energy source can be recharged. }
    \ReqItem{R:Charger}{The device includes a charger. }
    \ReqItem{R:Charger_enclosure}{The device can be charged inside the storage enclosure. }
    \ReqItem{R:Bat_Status}{An indicator provides information about the state of charge. }
    \ReqItem{R:Bat_Replace}{The internal energy source can be easily replaced using basic tools. Screwdrivers and pliers are considered basic tools. A soldering iron is considered a special tool, therefore no soldering must be required to replace the energy source. }
    \ReqItemMM{R:Chargingtime}{Charging time: }{}{$\qty{4}{\hour}$}
    \ReqItemMM{R:Endurance}{Runtime at maximum power output: }{$\qty{30}{\min}$}{}
    \ReqItem{R:60V}{Output mode to provide 60V}
    \ReqItemMM{R:60V_Tolerance}{Allowed deviation from target voltage in voltage mode: }{}{$\qty{1}{\percent}$}
    \ReqItemMM{R:60V_Current}{Output current in voltage mode: }{$\qty{1}{\ampere}$}{}
    \ReqItem{R:BSPD}{Output mode to provide an adjustable current}
    \ReqItemMM{R:BSPD_Current}{Maximum output current in current mode}{$\qty{10}{\ampere}$}{}
    \ReqItemMM{R:BSPD_Tolerance}{Allowed deviation from target output current in current mode: }{}{$\qty{1}{\percent}$}
    \ReqItemMM{R:BSPD_Voltage}{Maximum output voltage in current mode}{$\qty{5}{\volt}$}{}
    \ReqItem{R:Output}{The output terminals must be shrouded \qty{4}{\milli\meter} jacks. }
    \ReqItem{R:Calibration}{A means to calibrate the output values is provided. }
    \end{ReqTable}
\end{table}

\subsection{Safety requirements}
\begin{table}[h!]
    \begin{ReqTable}
    \ReqHeader
    \ReqItem{R:Iso}{The device must provide galvanic isolation between the charging port and the output terminals. }
    \ReqItem{R:Overvolt}{The device must not be permanently damaged when connected to \qty{600}{\volt}. }
    \ReqItem{R:Short}{A short circuit of the battery or the charging port must be prevented. }
    \ReqItem{R:Overcurrent}{An overcurrent protection for the battery must be included. }
    \ReqItem{R:BMS}{The battery must be protected against deep-discharge, over-charge and over-temperature. }
    \ReqItemMM{R:Rain}{The device must be able to operate in rain. The enclosure fulfills an ingress protection of at least: }{IP54}{}
    \end{ReqTable}
\end{table}
