\subsection{Safety requirements}
\begin{table}[h!]
    \begin{ReqTable}
    \ReqHeader
    \ReqItem{R:Iso}{Galvanic isolation between charging port and output}
    \ReqItem{R:Overvolt}{The device must not be permanently damaged when connected to \qty{600}{\volt}. }
    \ReqItem{R:Short}{A short circuit of the battery or the charging port must be prevented. }
    \ReqItem{R:Overcurrent}{An overcurrent protection for the battery must be included. }
    \ReqItem{R:BMS}{The battery must be protected against deep-discharge, over-charge and over-temperature. }
